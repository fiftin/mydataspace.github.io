Для импорта данных существует библиотека для NodeJS.

На данный момент это единственный способ импорта больших объемов данных в MyDataSpace. В дальнейшем будет возможность импорта данных из файлов с помощью Административной панели.

Ниже шаг за шагом описан процесс импорта данных на примере данных Geonames по почтовым индексам РФ.

1. Создаем рут и элемент для данных

Рут — корневой элемент девева данных.

Для удобства создадим дочерний элемент postal-codes в котовый и поместим данные из файла.

{{ 'https://myda.space' | replace: "https://","https://" }}

2. Создаем приложение

Чтобы наша NodeJS-программа могло сохранить данных в MyDataSpace, необходимо разрешить ей это через приложение.

Создадим новое приложение и разрешим доступ с любого IP адреса (для поля URLs & IPs/IP и URL адреса установить значение 0.0.0.0).

{{ 'https://myda.space' | replace: "https://","https://" }}

3. Скачиваем данные

Скачиваем и распаковываем файлы с данными.

4. Создаем NodeJS приложение

{% highlight bash %} mkdir import-geonames-data cd import-geonames-data npm init npm install mydataspace --save touch app.js {% endhighlight %}

app.js (не забудьте указать ACCESS_KEY, ACCESS_KEY, ROOT):

{% highlight javascript %} // Путь к файлу с данными const FILE_NAME = '/path/to/RU.txt'; // Ключ API вашего приложение const ACCESS_KEY = '***********'; // Корень в котором вы хотите сохранить данные const ROOT = 'geonames-demo'; const crypto = require('crypto'); const mds = require('mydataspace'); const myda = new mds.Myda({ root: ROOT, permission: ROOT, clientId: ACCESS_KEY }); const lineReader = require('readline').createInterface({ input: require('fs').createReadStream(FILE_NAME) }); var block = []; var linesRead = 0; /** * Connect to MyDataSpace and start importing data. */ myda.connect().then(() => myda.loginByToken(ACCESS_KEY)).then(function() { console.log('--- Logged in ---'); lineReader.on('line', parseLineAndAddToBlock); lineReader.on('pause', sendBlockToServer); // lineReader.on('resume', () => console.log('Resume')); lineReader.on('end', sendBlockToServer.bind(null, true)); }).catch(function(err) { console.log(err); }); function parseLineAndAddToBlock(line) { linesRead++; console.log(' parse ' + linesRead); const parts = line.split('\t'); const suffix = crypto.createHash('md5').update(line).digest('hex').substring(0, 7); block.push({ root: 'geonames', path: 'postal-codes/' + parts[0] + '-' + parts[1] + '-' + suffix, fields: [ { name: 'code', value: parts[1], type: 's' }, { name: 'placeName', value: parts[2], type: 's' }, { name: 'adminName1', value: parts[3], type: 's' }, { name: 'adminCode1', value: parts[4], type: 's' }, { name: 'adminName2', value: parts[5], type: 's' }, { name: 'adminCode2', value: parts[6], type: 's' }, { name: 'adminName3', value: parts[7], type: 's' }, { name: 'adminCode3', value: parts[8], type: 's' }, { name: 'latitude', value: parts[9], type: 's' }, { name: 'longitude', value: parts[10], type: 's' }, { name: 'accuracy', value: parts[11], type: 's' } ] }); if (block.length >= 1000) { lineReader.pause(); } } function nextLineOrExit(isLastBlock, isError) { if (isError) { if (isLastBlock) { console.log('*** Importing finished with error! ***'); process.exit(); } else { console.log('--- Block sent with errors. Reading new block... ---'); lineReader.resume(); } } else { if (isLastBlock) { console.log('*** Importing finished! ***'); process.exit(); } else { console.log('---Block sent successful. Reading new block... ---'); lineReader.resume(); } } } function sendBlockToServer(isLastBlock) { console.log('New block of ' + block.length + ' recoreds received. Sending it to server...'); var blockForSend = block; block = []; myda.request('entities.create', blockForSend, function() { nextLineOrExit(isLastBlock, false); }, function(err) { console.log('ERROR:'); console.log(err); console.log('INNER ERRORS:'); (Array.isArray(err) ? err : []).forEach(x => console.log(x.errors)); nextLineOrExit(isLastBlock, true); }); } {% endhighlight %}

5. Запускаем приложение

{% highlight bash %} node app {% endhighlight %}
